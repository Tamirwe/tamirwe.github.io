         function isBalanced(str) {
            str = str.replace(/[^\(\)]/g, '');
            var isMatch = false, result = "Balanced !";
            var matchIndex;

            var arr = str.split('');
            if (arr[0] == "(") {
                isMatch = true;

                while (arr.length > 0 && isMatch) {
                    matchIndex = checkBalance(arr);
                    if (matchIndex == 0) {
                        isMatch = false;
                    }
                    else {
                        arr.splice(matchIndex, 1)
                        arr.shift();
                    }
                }
            }

            return isMatch;

            function checkBalance(arr) {
                var matchNum = 0;
                var isOpen = false;

                for (var i = 0; i < arr.length; i++) {
                    if (arr[i] == '(') {
                        if (!isOpen) {
                            isOpen = true;
                        }
                        else {
                            matchNum++;
                        }
                    }
                    else if (isOpen && arr[i] == ')') {
                        if (matchNum == 0) {
                            return i;
                        }
                        else {
                            matchNum--;
                        }
                    }
                }
                return 0;
            }
        }

        private static int checkBalance(string str)
        {
            int matchNum = 0;
            bool isOpen = false;

            for (int i = 0; i < str.Length; i++)
            {
                if (str[i] == '(')
                {
                    if (!isOpen)
                    {
                        isOpen = true;
                    }
                    else
                    {
                        matchNum++;
                    }
                }
                else if (isOpen && str[i] == ')')
                {
                    if (matchNum == 0)
                    {
                        return i;
                    }
                    else
                    {
                        matchNum--;
                    }
                }
            }

function bubbleSort(items) {
  var length = items.length;
  for (var i = 0; i < length; i++) { //Number of passes
    for (var j = 0; j < (length - i - 1); j++) { //Notice that j < (length - i)
      //Compare the adjacent positions
      if(items[j] > items[j+1]) {
        //Swap the numbers
        var tmp = items[j];  //Temporary variable to hold the current number
        items[j] = items[j+1]; //Replace current number with adjacent number
        items[j+1] = tmp; //Replace adjacent number with current number
      }
    }        
  }
}
