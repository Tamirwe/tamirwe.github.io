Net Framework

is a platform that provides tools and technologies to develop Windows, Web and Enterprise applications.
It mainly contains two components,
1. Common Language Runtime (CLR)
2. .Net Framework Class Library.

CLR

provides an environment to run all the .Net Programs. The code which runs under the CLR is called as Managed Code.
Programmers need not to worry on managing the memory if the programs are running under the CLR as it provides memory management and thread management.
CLR manages the execution of programs written in different supported languages.
CLR transforms source code into a form of bytecode known as Common Intermediate Language (CIL).
At run time, CLR handles the execution of the CIL code.

JIT

When you compile your C# code in visual studio it gets turned into IL
During runtime the IL is interpreted into machine code and then it's executed.
This process is called Just In Time compilation or JIT for short.
Only the IL that is needed is transformed into machine code (and only once, it's "cached" once it's compiled into machinecode).

Check Parenthesis Balancing - javascript

         function isBalanced(str) {
            str = str.replace(/[^\(\)]/g, '');
            var isMatch = false, result = "Balanced !";
            var matchIndex;

            var arr = str.split('');
            if (arr[0] == "(") {
                isMatch = true;

                while (arr.length > 0 && isMatch) {
                    matchIndex = checkBalance(arr);
                    if (matchIndex == 0) {
                        isMatch = false;
                    }
                    else {
                        arr.splice(matchIndex, 1)
                        arr.shift();
                    }
                }
            }

            return isMatch;

            function checkBalance(arr) {
                var matchNum = 0;
                var isOpen = false;

                for (var i = 0; i < arr.length; i++) {
                    if (arr[i] == '(') {
                        if (!isOpen) {
                            isOpen = true;
                        }
                        else {
                            matchNum++;
                        }
                    }
                    else if (isOpen && arr[i] == ')') {
                        if (matchNum == 0) {
                            return i;
                        }
                        else {
                            matchNum--;
                        }
                    }
                }
                return 0;
            }
        }

Check Parenthesis Balancing - C#

        static void Main(string[] args)
        {
            string str = "(()())()((((())())())())";
            str = "()";
            str = "(()";
            bool isMatch = true;
            int matchIndex;
            StringBuilder sb = new StringBuilder(str);

            while (sb.Length > 0 && isMatch)
            {
                matchIndex = checkBalance(sb.ToString());
                if (matchIndex == 0)
                {
                    isMatch = false;
                }
                else
                {
                    sb.Remove(matchIndex, 1);
                    sb.Remove(0, 1);
                }
            }
        }

        private static int checkBalance(string str)
        {
            int matchNum = 0;
            bool isOpen = false;

            for (int i = 0; i < str.Length; i++)
            {
                if (str[i] == '(')
                {
                    if (!isOpen)
                    {
                        isOpen = true;
                    }
                    else
                    {
                        matchNum++;
                    }
                }
                else if (isOpen && str[i] == ')')
                {
                    if (matchNum == 0)
                    {
                        return i;
                    }
                    else
                    {
                        matchNum--;
                    }
                }
            }
            return 0;
        }
